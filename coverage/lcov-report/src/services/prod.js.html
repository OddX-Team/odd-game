import axios from 'axios'
import utils from '../utils'
 
const API = '/api'
 
function api (method, svc, data = {}) {
  const url = `${API}/${svc}`
 
  const params = {
    method,
    url,
    data,
    headers: {
      'Content-Type': 'application/json'
    }
  }
  return axios(params).then(resp => utils.camelizeKeys(resp.data))
}
